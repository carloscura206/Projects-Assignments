results_trainList = []
results_testList = []
for depth in range(1,50,5):
    np.random.seed(1000) 
    clf_dt = make_pipeline( StandardScaler(), tree.DecisionTreeClassifier(max_depth=depth))
    
    
    #
    accuracy_resultsTrain =  cross_val_score(clf_dt, X_train, y_train, cv=5, scoring = "accuracy")
    accuracy_resultsTest =  cross_val_score(clf_dt, X_test, y_test, cv=5, scoring = "accuracy")
    
    results_trainList.append([depth, 100 * np.mean(accuracy_resultsTrain)])
    
    
    results_testList.append([depth, 100 * np.mean(accuracy_resultsTest)])
    
    
df_results_list_dtTrain = pd.DataFrame(results_trainList, columns=["Max_Depth", "Acc"])
df_results_list_dtTest = pd.DataFrame(results_testList, columns=["Max_Depth", "Acc"])
